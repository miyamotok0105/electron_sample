publicPathを修正    

```js:webpack.config.js
module.exports = {
  entry: './src/main.js',
  output: {
    path: path.resolve(__dirname, './dist'),
    publicPath: '/', //修正
    filename: 'build.js'
  },
```


contentBase: 'dist',を追加     


```js:webpack.config.js
  devServer: {
    contentBase: 'dist', //追加
    historyApiFallback: true,
    noInfo: true,
    overlay: true
  },
  performance: {
    hints: false
  },
  devtool: '#eval-source-map'
}
```


再度実行


```
npm run build
firebase deploy
```
